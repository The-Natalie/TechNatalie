  • Instagram Feed Widget - A client wanted an Instagram feed on their home page that matched the Twitter feed that was already there. Unlike most social media platforms, Instagram doesn't provide a way to embed a feed(or multiple posts), they only provide embedded code for a single specified post. So, after explaining to the client how to retrieve their Instagram access token, Natalie used the plugin Instafeed.js to create the widget. Everything looked fine, except the images and text weren't posting. Turns out the client's firewall was blocking Instagram, which made Instafeed.js useless. So, Natalie's boss wrote a server script that calls the 12 most recent Instagram posts to the API, and puts it in a file on the server every 10 minutes. Natalie had to write new code that would parse through that information and output the appropriate HTML for the widget. She also updated the CSS to match the Twitter feed. Below is the final JavaScript code. The image to the right is the result.
    Languages used: HTML, CSS, JavaScript, jQuery, Mustache.js, Instafeed.js, Instagram's API 
    $(document).ready(function() {
  var template = '<span><a class="instagram-feed" href="{{{link}}}" target="_blank"><img src=
  "{{{images.low_resolution.url}}}" alt="{{caption.text}}"></a><p>{{caption.text}}</p></span>'
  Mustache.parse(template);
  var instaDiv = $('#instafeed');
    $.get("/instagram/feed.json")
      .done(function(response) {
        response.data.forEach(function(post) {
          var rendered = Mustache.render(template, post);
          instaDiv.append(rendered);
        })
      })
});


  • Turn Off Links with Pop-up - A client's website had multiple files that were updated quarterly, but sometimes the files would expire before they were able to update them. Instead of removing the links and code for hundreds of files only to have to rewrite them a few days later, a flag was created within in the links so that the link was disabled and a message would pop-up to let the visitor know that the file is being updated. If the "data-available" attribute within the link HTML code equaled "0", the link was disabled and a pop-up message was created, which came from the "data-message" attribute. As a backup, if the "data-message" attribute was left blank, there was a default message included in the JavaScript code. If the "data-available" attribute equaled "0", the link worked normally. The first code block below is an example of a link with the HTML code Natalie wrote. The second code block is the JavaScript Natalie wrote to make the flag work. The image to the right is an example of the pop-up message.
    Languages used: HTML, JavaScript, and jQuery 
    <a href="link/example.com" data-available="0" data-message="This piece is temporarily 
unavailable while it is being updated.">Link</a>
    function turnOffFiles() {
  $("a[data-available='0']").on("click", function(event) {
    event.preventDefault();
    event.stopPropagation();
    var dataMessage = $(event.currentTarget).data("message");
    var search_dialog = $( "#search-dialog" ).dialog({
      autoOpen: false,
      height: 300,
      width: 400,
      modal: true,
      buttons: {
        Close: function() {
          search_dialog.dialog( "close" );
        }
      },
      close: function() {}
          });

      $(".ui-dialog-title").html("Notification");
      search_dialog.dialog( "open" );
    if (typeof dataMessage !== 'undefined' && dataMessage !== '') {
       $('#search-dialog-results').html(dataMessage);
       search_dialog;
    }
     else {
     $('#search-dialog-results').html("This piece is not currently available.");
      search_dialog;
    }
  })
}

function setDuration() {
  var minutes = Math.floor(video.duration/ 60);
  var seconds = Math.floor(video.duration - (minutes * 60));
  if(seconds < 10) {
    seconds = '0'+seconds;
  }

$(document).ready(function() {
  // Video
  var video = document.getElementById("video");

  // Buttons
  var playButton = document.getElementById("pause");

  // Sliders
  var seekBar = document.getElementById("seek");
  var progressBar = document.getElementById("progress");

  $('.play').on('click', function(e) {
    $(this).fadeOut();
    $(this).siblings('.logo').fadeOut();
    $(this).siblings('.overlay').fadeOut();
    $('.slider').addClass('slide-down');
    if ('video') {
      $(this).siblings('video')[0].play();
      $('#controls').slideDown();
      $(playButton).addClass('icon-pause');
      $('#position').html('0:00');
      setDuration();
    }
  });
  $(seekBar).on('change', function() {
    // Calculate the new time
    var time = video.duration * (seekBar.value / 100);

    // Update the video time
    video.currentTime = time;
  });
  $(playButton).on('click', function(e) {
    $(this).toggleClass('icon-play icon-pause');
    if (video.paused == true) {
      // Play the video and fade out logo
      video.play();
      $(this).closest('.video').find('.logo').fadeOut();
    } else {
      // Pause the video and fade in logo
      video.pause();
      $(this).closest('.video').find('.logo').fadeIn();
    }
  });

  $(video).on('timeupdate', function(e) {
    // Calculate the slider value
    var value = (100 / video.duration) * video.currentTime;
    var minutes = Math.floor(video.currentTime/ 60);
    var seconds = Math.floor(video.currentTime - (minutes * 60));
    if(seconds < 10) {
      seconds = '0'+seconds;
    }
    $('#position').html(minutes+':'+seconds);

    // Update the slider value
    seekBar.value = value;
    progressBar.value = value;
  });
  $(video).on('ended', function(e) {
    $(this).siblings('.play').fadeIn();
    $(this).siblings('.logo').fadeIn();
    $(this).siblings('.overlay').fadeIn();
    $('.slider').removeClass('slide-down');
    $(playButton).removeClass('icon-pause');
    $('#controls').slideUp();
  });

  var fullScreenButton = document.getElementById('fullscreen');
  var theVideo = $('.video');

  /* Get into full screen */
  function GoInFullscreen(element) {
    if(element.requestFullscreen)
      element.requestFullscreen();
    else if(element.mozRequestFullScreen)
      element.mozRequestFullScreen();
    else if(element.webkitRequestFullscreen)
      element.webkitRequestFullscreen();
    else if(element.msRequestFullscreen)
      element.msRequestFullscreen();
  }

  /* Get out of full screen */
  function GoOutFullscreen() {
    if(document.exitFullscreen)
      document.exitFullscreen();
    else if(document.mozCancelFullScreen)
      document.mozCancelFullScreen();
    else if(document.webkitExitFullscreen)
      document.webkitExitFullscreen();
    else if(document.msExitFullscreen)
      document.msExitFullscreen();
  }

  /* Is currently in full screen or not */
  function IsFullScreenCurrently() {
    var full_screen_element = document.fullscreenElement || document.webkitFullscreenElement || document.mozFullScreenElement || document.msFullscreenElement || null;

    // If no element is in full-screen
    if(full_screen_element === null)
      return false;
    else
      return true;
  }

  // Show and hide full screen and minimize buttons
  $(fullScreenButton).on('click', function() {
    if(IsFullScreenCurrently()) {
      GoOutFullscreen();
      $(this).find(".go-to-fullscreen").show();
      $(this).find(".exit-fullscreen").hide();
    } else {
      GoInFullscreen(theVideo[0]);
      $(this).find(".exit-fullscreen").show();
      $(this).find(".go-to-fullscreen").hide();
    }
  });
});
